Retrieve invoices/usages from customers.


{% parameters get %}
 {% parameter 'task' 'Yes' 'Must be set to <em>usage</em> .' %}
 {% parameter 'number' 'Yes' 'Must be the invoice number you received from SendGrid.' %}
{% endparameters %}
